% ==================== PERFORMANCE MEASUREMENT ====================
\begin{frame}[t]{Performance Measurement and Attribution}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Return Calculation Methods}

\textbf{Time-Weighted Return (TWR):}
$$R_{TWR} = \prod_{t=1}^{n}(1 + R_t) - 1$$
\begin{itemize}
\item Removes cash flow impact
\item Standard for manager comparison
\end{itemize}

\textbf{Money-Weighted Return (MWR/IRR):}
\begin{itemize}
\item Includes timing of flows
\item Reflects client experience
\end{itemize}

% ==================== DIRECT INDEXING ====================
\begin{frame}[t]{Direct Indexing and Tax-Loss Harvesting}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{How Direct Indexing Works}
\begin{itemize}
\item Own individual securities (not fund)
\item Track index performance
\item Harvest tax losses daily
\item Apply ESG/custom screens
\end{itemize}

\vspace{0.3em}
\textbf{Tax Alpha Potential:}
$$\text{After-Tax Alpha} = T \times L \times r$$
Where:
\begin{itemize}
\item $T$ = tax rate (e.g., 37\%)
\item $L$ = harvested losses
\item $r$ = reinvestment return
\end{itemize}
