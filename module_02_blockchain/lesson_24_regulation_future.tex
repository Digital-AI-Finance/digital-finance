\begin{frame}{The Howey Test: Security vs Commodity}
\vspace{-2mm}
\textbf{Supreme Court Test (1946):} Four criteria for a security
\begin{enumerate}
\item Investment of money
\item In a common enterprise
\item With expectation of profits
\item Derived from efforts of others
\end{enumerate}

\vspace{3mm}
\textbf{Application to Crypto:}
\begin{itemize}
\item \textbf{Bitcoin:} Commodity (sufficiently decentralized, no central promoter)
\item \textbf{Ethereum:} Gray area (was security at ICO, now commodity per SEC)
\item \textbf{Most ICO tokens:} Securities (team-driven, profit expectation)
\item \textbf{XRP:} Lawsuit ongoing (Ripple vs SEC, mixed rulings)
\end{itemize}

\vspace{3mm}
\textbf{Implication:} Securities require registration, compliance costs prohibitive for many projects
\end{frame}

\begin{frame}{European Union: MiCA Regulation}
\vspace{-2mm}
\textbf{Markets in Crypto-Assets (MiCA):} Comprehensive framework (effective 2024)

\vspace{3mm}
\textbf{Key Provisions:}
\begin{itemize}
\item \textbf{Stablecoins:} Reserve requirements, redemption rights, daily transaction limits
\item \textbf{Crypto Exchanges:} Licensing, consumer protection, segregation of funds
\item \textbf{Token Issuers:} Whitepaper requirements, disclosures
\item \textbf{AML/KYC:} Travel Rule (identify transaction parties)
\item \textbf{DeFi:} Unclear, may require centralized interfaces to comply
\end{itemize}

\vspace{3mm}
\textbf{Impact:}
\begin{itemize}
\item Clarity for institutional adoption
\item Compliance costs favor established players
\item Potential stifling of innovation
\end{itemize}
\end{frame}

\begin{frame}{AML/KYC and the Travel Rule}
\vspace{-2mm}
\textbf{FATF Travel Rule (2019):}
\begin{itemize}
\item Virtual Asset Service Providers (VASPs) must share customer info for transactions $>$\$1000
\item Sender and receiver identity transmitted
\item Same as traditional banking wire transfers
\end{itemize}

\vspace{3mm}
\textbf{Challenges for Crypto:}
\begin{itemize}
\item Self-hosted wallets (DeFi, DEX) exempt (for now)
\item Privacy coins (Monero, Zcash) incompatible
\item Cross-jurisdictional enforcement difficult
\end{itemize}

\vspace{3mm}
\textbf{Industry Response:}
\begin{itemize}
\item Exchanges implement Travel Rule solutions (Chainalysis, Elliptic)
\item Privacy advocacy (coin mixing, zero-knowledge proofs)
\end{itemize}
\end{frame}

\begin{frame}{DeFi Regulation: The Frontier}
\vspace{-2mm}
\textbf{Regulatory Dilemma:}
\begin{itemize}
\item DeFi protocols are code (no company, no office)
\item DAO governance (token holders, pseudonymous)
\item Borderless, permissionless access
\item How to regulate code?
\end{itemize}

\vspace{3mm}
\textbf{Regulatory Strategies:}
\begin{itemize}
\item \textbf{Front-End Restrictions:} Geo-blocking, compliance layers (Uniswap blocks US users from interface)
\item \textbf{Developer Liability:} Tornado Cash sanctions (OFAC), developer arrested
\item \textbf{DAO as Legal Entity:} Wyoming DAO LLC, attempting legal clarity
\end{itemize}

\vspace{3mm}
\textbf{Debate:} Can truly decentralized protocols be regulated, or only centralized intermediaries?
\end{frame}

\begin{frame}{Philosophical Questions}
\vspace{-2mm}
\begin{itemize}
\item \textbf{Decentralization vs Efficiency:} Can we have both, or must we choose?
\item \textbf{Code is Law:} Should immutable contracts override legal recourse?
\item \textbf{Privacy vs Transparency:} How to balance anti-money laundering with individual privacy?
\item \textbf{Inequality:} Does crypto democratize finance or amplify wealth concentration?
\item \textbf{Environmental Impact:} Is PoW's energy use justified by its security?
\item \textbf{Governance:} Can token-based voting be truly democratic, or plutocratic?
\end{itemize}

\vspace{3mm}
\textbf{No Easy Answers:} These trade-offs define the ongoing evolution of blockchain technology
\end{frame}
