\begin{frame}{The Volatility Problem}
\vspace{-2mm}
\begin{center}
\includegraphics[width=0.48\textwidth]{figures/crypto_volatility/crypto_volatility.pdf}
\end{center}
\vspace{-3mm}
\textbf{Challenge:}
\begin{itemize}
\item BTC, ETH too volatile for everyday transactions
\item Cannot price goods, pay salaries, or lend in volatile assets
\item Need stable unit of account
\end{itemize}

\textbf{Solution:} Stablecoins -- cryptocurrencies pegged to stable assets (typically USD)
\bottomnote{Key concepts from this slide inform practical applications in finance.}
\end{frame}

\begin{frame}{What is a Stablecoin?}
\vspace{-2mm}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Definition:}
\begin{itemize}
\item Cryptocurrency designed to maintain stable value
\item Typically pegged 1:1 to fiat (USD, EUR)
\item Combines crypto benefits with price stability
\end{itemize}

\vspace{3mm}
\textbf{Use Cases:}
\begin{itemize}
\item Medium of exchange
\item Store of value (short-term)
\item DeFi collateral
\item Trading pairs (BTC/USDT)
\item Remittances
\end{itemize}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/stablecoin_use_cases/stablecoin_use_cases.pdf}
\end{columns}
\bottomnote{Clear definitions are essential for understanding complex technical concepts.}
\end{frame}

\begin{frame}{Stablecoin Taxonomy}
\vspace{-2mm}
\begin{center}
\includegraphics[width=0.48\textwidth]{figures/stablecoin_types/stablecoin_types.pdf}
\end{center}
\vspace{-3mm}
\textbf{Three Main Types:}
\begin{enumerate}
\item \textbf{Fiat-Collateralized:} Backed by USD reserves (USDT, USDC)
\item \textbf{Crypto-Collateralized:} Over-collateralized with crypto (DAI)
\item \textbf{Algorithmic:} No collateral, algorithmic supply adjustment (UST -- failed)
\end{enumerate}
\bottomnote{Stablecoins bridge traditional and crypto finance by maintaining price stability.}
\end{frame}

\begin{frame}{Fiat-Collateralized: Tether (USDT)}
\vspace{-2mm}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Mechanism:}
\begin{itemize}
\item 1 USDT = \$1 in bank reserves
\item Custodian holds fiat
\item Users trust issuer's solvency
\item Mint: Deposit \$1 $\rightarrow$ receive 1 USDT

\item Regulatory compliance

\begin{frame}{Tether Controversy: Are Reserves Real?}
\vspace{-2mm}
\textbf{Claims:}
\begin{itemize}
\item Tether claims 1:1 backing since 2014
\item Market cap: \$100B+ (2024)
\end{itemize}

\vspace{3mm}
\textbf{Controversies:}
\begin{itemize}
\item Lack of audits (only attestations, not full audits)
\item NYAG investigation (2021): Admitted reserves not fully in cash (commercial paper, crypto)
\item \$41M settlement, barred from New York trading
\item 2024 breakdown: $\sim$85\% T-bills, $\sim$10\% repo, $\sim$5\% other
\end{itemize}

\vspace{3mm}
\textbf{Risk:} Bank run scenario -- if everyone redeems simultaneously, can Tether honor?

\begin{frame}{Crypto-Collateralized: DAI (MakerDAO)}
\vspace{-2mm}
\begin{center}
\includegraphics[width=0.48\textwidth]{figures/dai_mechanism/dai_mechanism.pdf}
\end{center}
\vspace{-3mm}
\textbf{Mechanism:}
\begin{enumerate}
\item Deposit ETH (or other crypto) into Maker vault
\item Mint DAI (over-collateralized, e.g., 150\% ratio)
\item If collateral drops below threshold, liquidated
\item To reclaim collateral, repay DAI + stability fee (interest)
\end{enumerate}

\textbf{Decentralization:} No single custodian, governed by MKR token holders
\bottomnote{Cryptographic primitives provide the security foundation for blockchain systems.}
\end{frame}

\begin{frame}{Terra/Luna Mechanism: Arbitrage Peg}
\vspace{-2mm}
\textbf{Design:}
\begin{itemize}
\item 1 UST always redeemable for \$1 worth of LUNA (protocol guarantee)
\item Arbitrageurs maintain peg through minting/burning
\end{itemize}

\vspace{3mm}
\textbf{Case 1: UST $<$ \$1 (e.g., \$0.98)}
\begin{enumerate}
\item Buy 1 UST for \$0.98
\item Burn 1 UST $\rightarrow$ mint \$1 of LUNA
\item Sell LUNA for \$1
\item Profit: \$0.02 per UST (reduces UST supply, restores peg)
\end{enumerate}

\vspace{3mm}
\textbf{Case 2: UST $>$ \$1 (e.g., \$1.02)}
\begin{enumerate}
\item Burn \$1 of LUNA $\rightarrow$ mint 1 UST
\item Sell 1 UST for \$1.02
\item Profit: \$0.02 (increases UST supply, lowers price)
\end{enumerate}
\bottomnote{Key concepts from this slide inform practical applications in finance.}
\end{frame}

\begin{frame}{Why Did Terra Collapse?}
\vspace{-2mm}
\textbf{Structural Flaws:}
\begin{itemize}
\item \textbf{Ponzi Dynamics:} Anchor's 20\% APY funded by LUNA inflation (unsustainable)
\item \textbf{Circular Dependency:} UST backed by LUNA, LUNA value derived from UST demand
\item \textbf{No External Collateral:} Pure faith-based system
\item \textbf{Bank Run Vulnerability:} Confidence loss $\rightarrow$ redemptions $\rightarrow$ LUNA dilution $\rightarrow$ further confidence loss
\end{itemize}

\vspace{3mm}
\textbf{Trigger:}
\begin{itemize}
\item \$2B UST withdrawn from Anchor (possibly coordinated attack)
\item Luna Foundation Guard's \$3B BTC reserves insufficient to defend peg
\end{itemize}

\vspace{3mm}
\textbf{Aftermath:} \$60B market cap evaporated, lawsuits, Do Kwon (founder) arrested in Montenegro
\bottomnote{Key concepts from this slide inform practical applications in finance.}
\end{frame}

\begin{frame}{Lessons from Terra/Luna}
\vspace{-2mm}
\begin{enumerate}
\item \textbf{No Free Lunch:} 20\% risk-free yield is impossible sustainably
\item \textbf{Algorithmic Stablecoins are Fragile:} Require continuous growth, vulnerable to bank runs
\item \textbf{Circular Dependencies are Dangerous:} UST value depended on LUNA, LUNA on UST
\item \textbf{Reserves Matter:} \$3B BTC reserves inadequate for \$18B stablecoin
\item \textbf{Regulation Incoming:} Failures like Terra accelerate regulatory scrutiny
\item \textbf{Hubris Warning:} Do Kwon's overconfidence (``I don't debate the poor'') ignored critics
\end{enumerate}

\vspace{3mm}
\textbf{Quote:} ``If UST depegs, I will personally absorb the loss.'' -- Do Kwon (didn't happen)
\bottomnote{Key concepts from this slide inform practical applications in finance.}
\end{frame}

\textbf{Regulatory Push:}
\begin{itemize}

\item EU: MiCA regulation includes stablecoin provisions

\end{itemize}

\vspace{3mm}
\textbf{Market Shift:}
\begin{itemize}
\item Flight to regulated stablecoins (USDC, USDT)

\end{itemize}

\vspace{3mm}
\textbf{Innovation:}
\begin{itemize}
\item Frax: Hybrid model (partial collateral + algorithmic)
\item GHO (Aave): Over-collateralized, DAO-governed

\begin{frame}{Central Bank Digital Currencies (CBDCs)}
\vspace{-2mm}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Concept:}
\begin{itemize}
\item Government-issued digital currency
\item Not a stablecoin (is the fiat itself)
\item Centralized control
\item Programmable money
\end{itemize}

\vspace{3mm}
\textbf{Examples:}
\begin{itemize}
\item China: e-CNY (digital yuan)
\item Bahamas: Sand Dollar (launched 2020)
\item EU: Digital euro (pilot phase)
\item US: Digital dollar (research)
\end{itemize}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/cbdc_vs_stablecoin/cbdc_vs_stablecoin.pdf}
\end{columns}

\vspace{3mm}
\textbf{Concern:} Surveillance, programmable restrictions (e.g., expiry dates on money)
\bottomnote{Banks play a central role in the financial system as intermediaries.}
\end{frame}

\begin{frame}{Summary}
\vspace{-2mm}
\begin{itemize}
\item \textbf{Stablecoins:} Crypto with stable value, essential for DeFi and payments
\item \textbf{Fiat-Collateralized:} USDT/USDC, centralized, require trust in custodian
\item \textbf{Crypto-Collateralized:} DAI, over-collateralized, decentralized but capital-inefficient
\item \textbf{Algorithmic:} Terra/Luna, no collateral, death spiral vulnerability
\item \textbf{Terra Collapse:} \$60B lost, Ponzi dynamics (Anchor 20\%), circular dependency
\item \textbf{Lessons:} Unsustainable yields, importance of real reserves, regulatory response
\end{itemize}
