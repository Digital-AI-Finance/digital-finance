\begin{itemize}

\item Asset-Referenced Tokens (ARTs)
\item Other crypto-assets
\item Different regulatory tiers

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/mica_asset_types/mica_asset_types.pdf}

\column{0.48\textwidth}

\item Capital adequacy
\item Consumer protection
\end{itemize}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/mica_requirements/mica_requirements.pdf}
