\begin{frame}[t]{Strong Customer Authentication (SCA)}

\column{0.48\textwidth}
\textbf{Two-Factor Requirement}
\begin{itemize}
\item Knowledge (password)
\item Possession (mobile device)
\item Inherence (biometric)
\item Exemptions for low-risk
\end{itemize}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/sca_factors/sca_factors.pdf}
\end{columns}
\bottomnote{SCA requires 2 of 3 factors---knowledge, possession, or inherence---for secure authentication.}
