\begin{frame}{What is Fintech?}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Definition:}
\begin{itemize}
\item Financial Technology (``Fintech'')
\item Technology-enabled financial services innovation
\item Intersection of finance, technology, and business models
\item Digitization of financial services delivery
\end{itemize}

\vspace{5mm}
\textbf{Core Characteristics:}
\begin{itemize}
\item Customer-centric design
\item Digital-first approach
\item Data-driven decision making
\item Automation and efficiency
\end{itemize}

\column{0.48\textwidth}
\textbf{Not Just About Technology:}
\begin{itemize}
\item Business model innovation
\item Regulatory arbitrage
\item User experience transformation
\item Market structure disruption
\end{itemize}

\vspace{5mm}
\textbf{Evolution of Term:}
\begin{itemize}
\item Pre-2000s: Back-office technology
\item 2000s-2010: Internet banking, e-payments
\item 2010s: Startup-driven disruption
\item 2020s: Embedded finance, DeFi
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Historical Evolution of Fintech}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Era 1 (1866-1967): Foundation}
\begin{itemize}
\item 1866: First transatlantic cable
\item 1918: Fedwire launch
\item 1950: First credit card (Diners Club)
\item 1967: First ATM (Barclays, London)
\end{itemize}

\vspace{3mm}
\textbf{Era 2 (1967-2008): Digitization}
\begin{itemize}
\item 1973: SWIFT established
\item 1983: Bloomberg Terminal
\item 1994: First online banking
\item 1998: PayPal founded
\item 2007: First iPhone launched
\end{itemize}

\column{0.48\textwidth}
\textbf{Era 3 (2008-Present): Disruption}
\begin{itemize}
\item 2008: Global Financial Crisis
\item 2009: Bitcoin whitepaper
\item 2011: Stripe, Square growth
\item 2013: PSD2 directive (EU)
\item 2015: Neobank explosion
\item 2020: Embedded finance rise
\item 2023: Generative AI in finance
\end{itemize}

\vspace{3mm}
\textbf{Key Inflection Point:}
\begin{itemize}
\item Post-2008 crisis: Trust erosion in traditional banks
\item Smartphone adoption enabling mobile-first services
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Key Drivers of Fintech Growth}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Technology Enablers:}
\begin{itemize}
\item Cloud computing infrastructure
\item Mobile device ubiquity (5B+ smartphones)
\item API economy and microservices
\item Artificial intelligence and machine learning
\item Blockchain and distributed ledgers
\end{itemize}

\vspace{3mm}
\textbf{Regulatory Drivers:}
\begin{itemize}
\item Open banking mandates (PSD2, UK CMA)
\item Sandbox frameworks (FCA, MAS, FINMA)
\item Digital identity standards (eIDAS)
\item Licensing innovation (e-money, payment institutions)
\end{itemize}

\column{0.48\textwidth}
\textbf{Market Forces:}
\begin{itemize}
\item Millennial and Gen Z preferences
\item Underbanked populations (1.7B globally)
\item COVID-19 acceleration of digital adoption
\item Lower customer acquisition costs
\item Network effects and platform economics
\end{itemize}

\vspace{3mm}
\textbf{Economic Factors:}
\begin{itemize}
\item Low interest rates driving VC investment
\item Cost pressure on incumbents
\item Globalization of financial services
\item Declining transaction costs
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Fintech Categories: Payments}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Digital Payments:}
\begin{itemize}
\item Payment processors (Stripe, Adyen, Square)
\item Mobile wallets (Apple Pay, Google Pay, Alipay)
\item Cross-border transfers (Wise, Remitly, WorldRemit)
\item Buy-now-pay-later (Klarna, Affirm, Afterpay)
\end{itemize}

\vspace{3mm}
\textbf{Market Dynamics:}
\begin{itemize}
\item Shift from cash to digital (80\% of transactions by 2025)
\item Real-time payment networks (FedNow, SEPA Instant)
\item Embedded payments in non-financial apps
\item Cryptocurrency payment rails
\end{itemize}

\column{0.48\textwidth}
\textbf{Key Innovations:}
\begin{itemize}
\item QR code payments (WeChat Pay, Paytm)
\item Contactless NFC technology
\item Request-to-pay infrastructure
\item Account-to-account (A2A) payments
\end{itemize}

\vspace{3mm}
\textbf{European Context:}
\begin{itemize}
\item PSD2 enabling third-party access
\item Strong Customer Authentication (SCA)
\item Instant SEPA Credit Transfer (SCT Inst)
\item Digital euro development (ECB)
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Fintech Categories: Banking and Lending}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Neobanks/Challenger Banks:}
\begin{itemize}
\item Mobile-first current accounts
\item No physical branches
\item Examples: Revolut, N26, Monzo, Chime
\item EU: 35M+ neobank customers (2024)
\end{itemize}

\vspace{3mm}
\textbf{Alternative Lending:}
\begin{itemize}
\item Peer-to-peer lending (Funding Circle, LendingClub)
\item SME financing (Kabbage, OnDeck)
\item Student loans (SoFi, Earnest)
\item Invoice financing (BlueVine, Fundbox)
\end{itemize}

\column{0.48\textwidth}
\textbf{Business Models:}
\begin{itemize}
\item Freemium basic accounts
\item Premium subscriptions (\EUR5-15/month)
\item Interchange revenue from card spending
\item FX margin on currency exchange
\item Embedded finance partnerships
\end{itemize}

\vspace{3mm}
\textbf{Technology Advantages:}
\begin{itemize}
\item Lower cost-to-serve (CAC \$10-50 vs \$200-400)
\item Instant account opening (3-5 minutes)
\item AI-driven underwriting
\item Real-time spending analytics
\end{itemize}
\end{columns}
\end{frame}

\vspace{3mm}
\textbf{Investment Products:}
\begin{itemize}

\item ESG/thematic investing
\item Tax-loss harvesting
\item Cryptocurrency access
\item Alternative investments (Private equity, real estate)
\end{itemize}
\end{columns}
\end{frame}

\vspace{3mm}
\textbf{Enabling Technologies:}
\begin{itemize}

\item Machine learning for anomaly detection
\item Graph analytics for network analysis

\end{columns}
\end{frame}

\begin{frame}{Fintech vs Traditional Banks: Comparison}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Cost Structure:}
\begin{itemize}
\item Traditional bank cost-to-income: 55\%-65\%
\item Fintech cost-to-income: 35\%-50\%
\item Branch network vs cloud infrastructure
\item Legacy IT systems vs modern tech stack
\end{itemize}

\vspace{3mm}
\textbf{Customer Acquisition:}
\begin{itemize}
\item Bank CAC: \$200-\$400
\item Fintech CAC: \$10-\$50
\item Viral growth and referrals
\item Digital-native marketing
\end{itemize}

\column{0.48\textwidth}
\textbf{Competitive Advantages (Fintech):}
\begin{itemize}
\item Speed: Account opening in minutes vs days
\item UX: Mobile-first, intuitive design
\item Pricing: Lower fees, better FX rates
\item Innovation: Rapid product iteration
\end{itemize}

\vspace{3mm}
\textbf{Competitive Advantages (Banks):}
\begin{itemize}
\item Trust and brand recognition
\item Full banking license and deposit insurance
\item Diversified revenue streams
\item Regulatory expertise
\item Physical presence for complex needs
\end{itemize}
\end{columns}
\end{frame}

\item Customer retention and churn
\item Scaling while maintaining unit economics
\item Competition from Big Tech
\item Regulatory compliance costs
\end{itemize}

\vspace{3mm}
\textbf{Technology Risks:}
\begin{itemize}
\item Cybersecurity threats
\item System outages and downtime
\item Third-party vendor dependency
\item Data privacy concerns
\item AI bias and fairness
\end{itemize}

\column{0.48\textwidth}
\textbf{Regulatory Risks:}
\begin{itemize}
\item Licensing requirements
\item Capital adequacy rules
\item Consumer protection mandates
\item Cross-border complexity
\item Changing regulatory landscape
\end{itemize}

\vspace{3mm}
\textbf{Market Risks:}
\begin{itemize}
\item Interest rate sensitivity
\item Credit risk in lending models
\item Liquidity management
\item Valuation corrections (2022-2023)

\end{itemize}
\end{columns}
\end{frame}
