\section{Legacy vs Modern Architectures}

\begin{frame}[t]{Legacy Mainframe Systems}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Characteristics}
\begin{itemize}
\item COBOL/PL/I languages
\item Batch processing overnight
\item Monolithic architecture
\item Green-screen interfaces
\item Decades-old codebase
\item High operational cost
\end{itemize}

\column{0.48\textwidth}
\textbf{Challenges}
\begin{itemize}
\item Talent shortage (COBOL)
\item Inflexible product launch
\item Integration complexity
\item Vendor lock-in
\item Maintenance burden
\item Regulatory adaptation
\end{itemize}
\end{columns}

\vspace{1em}
\textbf{Reality}: 70\% of global banks still run mainframes (2024), processing 80\%+ transactions
\bottomnote{AI and ML are transforming financial services through automation and prediction.}
\end{frame}

\begin{frame}[t]{Why Legacy Systems Persist}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Strengths}
\begin{itemize}
\item Proven reliability
\item Transaction throughput
\item Data consistency
\item Security track record
\item Regulatory compliance
\item Sunk cost investment
\end{itemize}

\column{0.48\textwidth}
\textbf{Migration Risks}
\begin{itemize}
\item Business disruption
\item Data integrity issues
\item Customer impact
\item Regulatory approval
\item Cost overruns (2-3x budget)
\item Timeline delays (5-7 years)
\end{itemize}
\end{columns}

\vspace{1em}
\textbf{Case Study}: TSB migration disaster (2018), 1.9M customers affected, \pounds 330M cost
\bottomnote{Key concepts from this slide inform practical applications in finance.}
\end{frame}

\section{Modernization Strategies}

\begin{frame}[t]{Core Banking Modernization Approaches}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Big Bang Replacement}
\begin{itemize}
\item Complete system swap
\item Single cutover weekend
\item High risk, high reward
\item 12-36 months timeline
\item Example: Metro Bank (2010)
\end{itemize}

\column{0.48\textwidth}
\textbf{Gradual Migration}
\begin{itemize}
\item Product-by-product
\item Run legacy in parallel
\item Lower risk
\item 5-7 years timeline
\item Example: DBS Bank
\end{itemize}
\end{columns}

\vspace{1em}
\textbf{Hybrid}: New products on modern core, migrate legacy gradually
\bottomnote{Banks play a central role in the financial system as intermediaries.}
\end{frame}

\begin{frame}[t]{Strangler Fig Pattern}
\textbf{Strategy}: Incrementally replace legacy by routing new functionality to modern system

\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Steps}
\begin{enumerate}
\item API facade over legacy
\item Route new features to modern
\item Migrate data incrementally
\item Decommission legacy modules
\item Repeat until complete
\end{enumerate}

\column{0.48\textwidth}
\textbf{Advantages}
\begin{itemize}
\item Continuous business operation
\item Lower risk profile
\item Incremental investment
\item Learning by doing
\item Easier rollback
\end{itemize}
\end{columns}

\vspace{1em}
\textbf{Duration}: 3-5 years typical, allows parallel operation
\bottomnote{Key concepts from this slide inform practical applications in finance.}
\end{frame}

\begin{frame}[t]{Data Migration Challenges}
\textbf{Complexity}: 30-40 years of customer data, multiple systems, format inconsistencies

\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Data Issues}
\begin{itemize}
\item Schema mismatches
\item Data quality problems
\item Historical transactions
\item Regulatory retention
\item Customer consent
\end{itemize}

\column{0.48\textwidth}
\textbf{Migration Techniques}
\begin{itemize}
\item ETL pipelines
\item Data validation
\item Reconciliation
\item Parallel run testing
\item Rollback procedures
\end{itemize}
\end{columns}

\vspace{1em}
\textbf{Best Practice}: Migrate 80\% data, keep 20\% in archive for queries
\bottomnote{Understanding limitations helps identify appropriate use cases and avoid over-engineering.}
\end{frame}
