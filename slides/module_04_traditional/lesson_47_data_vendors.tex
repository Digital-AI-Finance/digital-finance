\begin{frame}[t]{Vendor Business Models}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Terminal-Based Model (Bloomberg):}
\begin{itemize}
\item \textbf{Pricing:} \$24-27k per terminal per year
\item \textbf{Users:} 350,000+ terminals globally (2024)
\item \textbf{Revenue:} \$12B+ annually (95\%+ from terminals)
\item \textbf{Value Prop:} Integrated platform (data + analytics + communication)
\item \textbf{Lock-In:} Workflow integration, chat network
\end{itemize}

\vspace{2mm}
\textbf{Enterprise Data Licensing (Refinitiv, FactSet):}
\begin{itemize}
\item \textbf{Pricing:} Per-user or enterprise-wide licenses
\item \textbf{Customization:} APIs, feeds, data integration
\item \textbf{Flexibility:} Choose specific datasets
\item \textbf{Cost:} \$10-50k+ per user depending on entitlements
\end{itemize}

\column{0.48\textwidth}
\textbf{Exchange Data Licensing:}
\begin{itemize}
\item \textbf{Direct from Exchanges:} NYSE, Nasdaq, CME, Eurex
\item \textbf{Professional vs Non-Professional:} Different fee tiers
\item \textbf{Display vs Non-Display:} Machine-only usage often higher fees
\item \textbf{Redistribution:} Additional fees for reselling data
\end{itemize}

\vspace{2mm}
\textbf{Alternative Data (Emerging):}
\begin{itemize}
\item \textbf{Pricing:} Subscription (monthly/annual) or per-use
\item \textbf{Range:} \$10k-500k+ annually depending on dataset
\item \textbf{Examples:} Satellite imagery, credit card data, web scraping
\item \textbf{Providers:} Specialty vendors (not traditional giants)
\end{itemize}

\vspace{2mm}
\textit{Typical hedge fund data spending: \$1-5M/year; large investment bank: \$50-100M+/year}
\end{columns}

\section{Data Licensing and Exchange Fees}

\begin{frame}[t]{Exchange Data Licensing Models}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Professional vs Non-Professional:}

\textbf{Non-Professional:}
\begin{itemize}
\item Individual investors (not paid for trading)
\item Lower fees: \$1-5/month per exchange
\item Examples: Retail brokerage users, students
\end{itemize}

\textbf{Professional:}
\begin{itemize}
\item Anyone compensated for securities-related work
\item Higher fees: \$50-200/month per exchange
\item Institutional traders, analysts, advisors
\end{itemize}

\vspace{2mm}
\textbf{Display vs Non-Display:}

\textbf{Display:}
\begin{itemize}
\item Human consumption (screens, reports)
\item Priced per user per month
\end{itemize}

\textbf{Non-Display:}
\begin{itemize}
\item Algorithmic trading, risk systems
\item Enterprise-level pricing (often higher than display)
\item Audit rights for exchanges (verify usage)
\end{itemize}

\column{0.48\textwidth}
\textbf{Depth of Book Fees:}

\textbf{Level 1 (Top of Book):}
\begin{itemize}
\item Best bid/ask only
\item Lowest fees: \$5-20/month (professional)
\end{itemize}

\textbf{Level 2 (Full Book):}
\begin{itemize}
\item All price levels and depth
\item Higher fees: \$50-150/month
\end{itemize}

\textbf{Level 3 (Order-by-Order):}
\begin{itemize}
\item Individual order IDs and attribution
\item Premium fees: \$500-2000/month
\item Used for market making and order flow analysis
\end{itemize}

\vspace{2mm}
\textbf{Enterprise Fees (Non-Display):}
\begin{itemize}
\item Flat fees: \$50k-500k+/year per exchange
\item Query-based: \$0.01-1.00 per query
\item Device-based: Fee per server/instance
\item Redistribution: Additional fees for reselling data
\end{itemize}
\end{columns}
\bottomnote{Quality data is the foundation for effective machine learning models.}
\end{frame}

\begin{frame}[t]{Vendor Data Redistribution Agreements}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Vendor Licensing Arrangements:}

\textbf{Bloomberg, Refinitiv, FactSet:}
\begin{itemize}
\item Bundle exchange fees into subscription
\item Negotiate enterprise agreements with exchanges
\item Pass-through fees or inclusive pricing
\item Simplifies client billing (single invoice)
\end{itemize}

\vspace{2mm}
\textbf{Controlled vs Derived Data:}

\textbf{Controlled (Raw):}
\begin{itemize}
\item Real-time prices, order book data
\item Subject to exchange fees and terms
\item Cannot redistribute without exchange permission
\end{itemize}

\textbf{Derived (Analytics):}
\begin{itemize}
\item Calculations based on raw data (indices, signals)
\item More flexible licensing (vendor IP)
\item Still may require exchange approval for some uses
\end{itemize}

\column{0.48\textwidth}
\textbf{Data Licensing Challenges:}

\textbf{Compliance and Audits:}
\begin{itemize}
\item Exchanges audit vendor and end-user compliance
\item User counts, usage types (display/non-display)
\item Penalties for under-reporting: 3-5x back fees
\item Automated monitoring tools (e.g., Vela, MDDM)
\end{itemize}

\textbf{Cost Optimization:}
\begin{itemize}
\item Audit internal user entitlements
\item Deactivate unused subscriptions (15-30\% savings typical)
\item Negotiate enterprise caps with vendors
\item Shift to derived data where permissible
\end{itemize}

\vspace{2mm}
\textbf{Regulatory Data (Free/Cheap):}
\begin{itemize}
\item \textbf{SEC EDGAR:} US public company filings
\item \textbf{FINRA TRACE:} Corporate bond trades
\item \textbf{Treasury.gov:} US government bonds
\item \textbf{ESMA FIRDS:} EU financial instruments reference
\end{itemize}
\end{columns}
\bottomnote{Quality data is the foundation for effective machine learning models.}
\end{frame}

\section{Data Quality and Governance}

\begin{frame}[t]{Data Quality Challenges}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Common Data Issues:}

\textbf{1. Stale or Missing Data:}
\begin{itemize}
\item Delayed pricing for illiquid securities
\item Corporate actions not reflected timely
\item Estimates not updated (analyst departures)
\end{itemize}

\textbf{2. Incorrect Reference Data:}
\begin{itemize}
\item Wrong ISINs or identifiers
\item Inaccurate corporate hierarchies
\item Currency mismatches
\end{itemize}

\textbf{3. Survivorship Bias:}
\begin{itemize}
\item Delisted companies removed from history
\item Overstates backtest performance
\item Critical for quant research
\end{itemize}

\textbf{4. Point-in-Time Issues:}
\begin{itemize}
\item Historical data reflects current state (not as-of)
\item Restatements not handled properly
\item Look-ahead bias in backtests
\end{itemize}

\column{0.48\textwidth}
\textbf{Data Governance Framework:}

\textbf{1. Data Validation:}
\begin{itemize}
\item Automated reasonableness checks (outliers)
\item Cross-vendor comparison (Bloomberg vs Refinitiv)
\item Manual review for critical datasets
\item Threshold alerts (price moves > 10\%)
\end{itemize}

\textbf{2. Data Lineage:}
\begin{itemize}
\item Track data source and transformations
\item Version control for datasets
\item Audit trail for regulatory compliance

\begin{frame}[t]{Cloud and API-First Architectures}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}

\textbf{Vendor Cloud Platforms:}
\begin{itemize}

\textbf{Benefits:}
\begin{itemize}
\item No on-premise infrastructure
\item Elastic scaling (pay per use)
\item Faster time-to-market
\item Global availability and redundancy
\end{itemize}

\textbf{Consumption-Based Pricing:}
\begin{itemize}
\item Pay per API call or data volume
\item Shift from per-user to usage-based
\item Better aligns cost with value
\item Challenges: Unpredictable costs, audit complexity
\end{itemize}

\column{0.48\textwidth}
\textbf{API Standardization:}

\textbf{OpenFIGI (Bloomberg):}
\begin{itemize}
\item Financial Instrument Global Identifier
\item Free API for security identification
\item Maps ISINs, CUSIPs, tickers to FIGI
\end{itemize}

\textbf{FINOS (Fintech Open Source):}
\begin{itemize}
\item Open-source financial data standards
\item Common Data Model (CDM) collaboration
\item Interoperability across vendors
\end{itemize}

\vspace{2mm}
\textbf{Real-Time Streaming:}
\begin{itemize}
\item WebSocket APIs for live data
\item Kafka-based message buses
\item Sub-second latency for cloud delivery
\item Use case: Real-time dashboards, algos
\end{itemize}

\vspace{2mm}
\textit{Major shift: Terminal-based $\rightarrow$ API-first data consumption (especially for quant firms)}
\end{columns}
