\section{Unit Economics}
% =============================================================================

\begin{frame}[t]{Unit Economics Deep Dive}
\begin{center}
\includegraphics[width=0.60\textwidth]{figures/fintech_unit_economics/fintech_unit_economics.pdf}
\end{center}
\bottomnote{Unit economics drive FinTech viability---CAC, LTV, payback period, and contribution margin are key metrics.}
\end{frame}

\begin{frame}[t]{Customer Lifetime Value (LTV)}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{LTV Calculation}
\begin{itemize}
\item Average revenue per user
\item Churn rate
\item Gross margin
\item LTV = ARPU $\times$ Margin / Churn
\end{itemize}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/ltv_calculation/ltv_calculation.pdf}
\end{columns}
\bottomnote{LTV = ARPU x Margin / Churn---maximizing LTV requires increasing revenue, margins, and retention.}
\end{frame}

\begin{frame}[t]{LTV/CAC Ratio Benchmarks}
\begin{columns}[T]
\scriptsize
\column{0.48\textwidth}
\textbf{Health Metrics}
\begin{itemize}
\item $<$ 1: Unprofitable growth
\item 1-3: Breakeven zone
\item 3+: Sustainable business
\item Best-in-class: 5-7x
\end{itemize}

\column{0.48\textwidth}
\includegraphics[width=\textwidth]{figures/ltv_cac_benchmarks/ltv_cac_benchmarks.pdf}
\end{columns}
\bottomnote{LTV/CAC > 3 indicates sustainable business---below 1 means unprofitable growth, best-in-class hit 5-7x.}
\end{frame}
